import rclpy
from rclpy.node import Node
from geometry_msgs.msg import TwistStamped

FORWARD = 0.1
SPIN = 0.15


class DriveRobot(Node):
    def __init__(self):
        super().__init__('drive_robot')

        self.publisher_ = self.create_publisher(TwistStamped, 'cmd_vel', 10)
        self.timer = self.create_timer(0.1, self.send_velocity)

        # Track elapsed time
        self.elapsed_time = 0.0
        self.stop_time = 3.0   # seconds

        self.spinning = False

    def send_velocity(self):
        msg = TwistStamped()
        msg.header.stamp = self.get_clock().now().to_msg()

        # Increase time counter
        if self.spinning:
            self.elapsed_time += SPIN
        else:
            self.elapsed_time += FORWARD

        if self.elapsed_time < self.stop_time and  not self.spinning:
            msg.twist.linear.x = -0.26
            msg.twist.angular.z = 0.0
        elif self.elapsed_time < self.stop_time and self.spinning:
            msg.twist.linear.x = 0.0
            msg.twist.angular.z = 1.82
        else:
            self.spinning = not self.spinning
            msg.twist.linear.x = 0.0
            msg.twist.angular.z = 0.0
            self.get_logger().info("Robot stopped")
            self.elapsed_time = 0.0

        self.publisher_.publish(msg)
   
        



def main():
    rclpy.init()
    node = DriveRobot()
    rclpy.spin(node)
    node.destroy_node()
    rclpy.shutdown()


if __name__ == '__main__':
    main()
